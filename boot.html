 73 ;读取硬盘为两种模式
 74 ;1. chs
 75 ;2. lba
 76 ;使用28byte表示逻辑扇区号
 77 ;端口号为 0x1f3~0x1f6
 78 ;从低位与端口号对应
 79 ;0x1f2 端口号 决定读多少扇区数 1~255 如果是0代表读256个扇区
 80 ;扇区号 = 0000   0000 0000   0000 0000 0000 0000
 81 ;       |0x1f6| | 0x1f5 |   | 0x1f4 | | 0x1f3  |
 82 ;其中0x1f6位8位端口 低四位用来表示lba 27~24
 83 ;高四位的 第4位0表示主硬盘，1表示从硬盘
 84 ;第5，7位固定为1
 85 ;第6位 0表示CHS模式，1表示LBA模式
 86 ;0x1f7 为命令&状态端口 是一个8位的端口
 87 ;0x20表示读
 88 ;0x1f7的详细信息 https://blog.csdn.net/cosmoslife/article/details/9024659
 89 ;0x1f0为数据端口
 90 
 91 read_hard_disk_0: ;从硬盘读取一个逻辑扇区
 92     push ax       ;输入 DI:SI=起始逻辑扇区号
 93     push bx       ;     DS:BX=目标缓冲区地址
 94     push cx
 95     push dx
 96 
 97     mov dx,0x1f2 ;0x1f2端口号为决定读取扇区数的
 98     mov al,1
 99     out dx,al ;读取扇区数
100 
101     inc dx    ;0x1f3端口
102     mov ax,si
103     out dx,al  ;LBA地址7~0
104 
105     inc dx      ;0x1f4端口:
106     mov al,ah   ;LBA地址 8~15
107     out dx,al
108 
109     inc dx      ;0x1f5端口
110     mov ax,di   ;LAB地址 16~23
111     out dx,al
112 
113     inc dx      ;0x1f6端口
114     mov al,0xe0 ;看上边注释
115     or al,ah
116     out dx,al
117 
118     inc dx      ;0x1f7端口,设置为读
119     mov al,0x20
120     out dx,al
121 
122 .waits:
123     in al,dx    ;获取disk状态，直到硬盘准备就绪
124     and al,0x88
125     cmp al,0x08
126     jnz .waits
127 
128     mov cx,256 ;一个扇区256个字 512个字节,每次读两个字节，1个字
129     mov dx,0x1f0
130 
131 .readw:
132     in ax,dx  ;读完
133     mov [bx],ax
134     add bx,2
135     loop  .readw
136 
137     pop dx
138     pop cx
139     pop bx
140     pop ax
141 
142     ret
